pub fn vec_extend() {
   let mut vec = std::vec::Vec::new();
   vec.push(1);
   assert_eq!(vec[0], 1);
   vec[0] = 7;
   assert_eq!(vec[0], 7);
   vec.extend([1, 2, 3]);
   assert_eq!(vec, [7, 1, 2, 3]);
}
vec_extend();
